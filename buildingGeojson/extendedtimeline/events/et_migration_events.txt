# Migration start
country_event = {
	id = et.29
	title = "et.EVTNAME29"
	desc = "et.EVTDESC29"
	picture = BORDER_TENSION_eventPicture

	trigger = {
		government = barbarian_tribe
		mil_tech = 15
		is_subject = no
		is_at_war = no
		NOT = { has_country_modifier = barbarian_migration }
		NOT = { has_country_flag = migration_starter }
		OR = {
			is_neighbor_of = ROM
			is_neighbor_of = ROW
			is_neighbor_of = BYZ
		}
	}
	mean_time_to_happen = {
		months = 12
	}
	
	option = {
		name = "et.EVTOPTA29"
		ai_chance = { 
			factor = 100

			modifier = {
				factor = 0
				NOT = { stability = 0 }
			}
			modifier = {
				factor = 0
				num_of_revolts = 1
			}
		}
		add_prestige = 20

		add_country_modifier = {
			name = barbarian_migration
			duration = 3560
		}
		if = {
			limit = { is_neighbor_of = ROM }
			declare_war = ROM
		}
		if = {
			limit = { is_neighbor_of = ROW NOT = { is_neighbor_of = ROM } }
			declare_war = ROW
		}
		if = {
			limit = { is_neighbor_of = BYZ NOT = { is_neighbor_of = ROW } NOT = { is_neighbor_of = ROM } }
			declare_war = BYZ
		}
		random_owned_province = {
			infantry = ROOT
			infantry = ROOT
			infantry = ROOT
			infantry = ROOT
			infantry = ROOT
			infantry = ROOT
			infantry = ROOT
			infantry = ROOT
		}
	}
	option = {
		name = "et.EVTOPTB29"
		ai_chance = { 
			factor = 1
		}
		add_prestige = -20
		set_country_flag = migration_starter
	}
}
